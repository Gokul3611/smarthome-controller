    const scanBtn = document.getElementById('scanBtn');
    const netList = document.getElementById('netList');
    const scanStatus = document.getElementById('scanStatus');
    const spinner = document.getElementById('scanSpinner');
    const ssidInput = document.getElementById('ssidInput');
    const pwdInput = document.getElementById('pwdInput');
    const resultMsg = document.getElementById('resultMsg');
    const provForm = document.getElementById('provForm');

    async function scanNetworks(){
      spinner.style.display = 'inline-block';
      scanStatus.textContent = 'scanning...';
      netList.innerHTML = '';
      try {
        const res = await fetch('/scan', {cache:'no-cache'});
        if(!res.ok) throw new Error('scan failed');
        const list = await res.json();
        if(!Array.isArray(list) || list.length===0){
          scanStatus.textContent = 'no networks found';
        } else {
          scanStatus.textContent = list.length + ' networks';
        }
        list.sort((a,b)=>b.rssi-a.rssi);
        list.forEach(n=>{
          const li = document.createElement('li');
          li.className = 'net';
          li.tabIndex = 0;
          const left = document.createElement('div');
          left.className = 'left';
          const mid = document.createElement('div');
          mid.innerHTML = `<div class="ssid">${escapeHtml(n.ssid)}</div><div class="meta">${Math.abs(n.rssi)} dBm</div>`;
          const right = document.createElement('div');
          right.className = 'signal';
          right.innerHTML = n.open ? '&#128275;' : signalBars(Math.abs(n.rssi));
          left.appendChild(mid);
          li.appendChild(left);
          li.appendChild(right);

          li.addEventListener('click', ()=> {
            ssidInput.value = n.ssid;
            pwdInput.focus();
          });
          netList.appendChild(li);
        });
      } catch(e){
        scanStatus.textContent = 'scan error';
      } finally {
        spinner.style.display = 'none';
      }
    }

    function signalBars(rssi){
      // simple bars using unicode blocks
      if(rssi < 40) return '&#127993;'; // 1 bar
      if(rssi < 60) return '&#127994;'; // 2
      if(rssi < 80) return '&#127995;'; // 3
      return '&#127996;'; // 4
    }

    function escapeHtml(s){ return String(s||'').replace(/[&<>"']/g, c=>({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c])); }

    scanBtn.addEventListener('click', scanNetworks);
    // auto-scan on load
    window.addEventListener('load', ()=>{ setTimeout(scanNetworks, 400); });

      ev.preventDefault();

          const t = await resp.text();
          resultMsg.textContent = 'Save failed: ' + t;
        }
      }catch(e){
        resultMsg.textContent = 'Network error';
      }
    });

#include "device_control.h"

static DeviceState devices[4];

void device_control_init() {
  for (int i = 0; i < 4; i++) {
    devices[i].type = DEVICE_LIGHT;
    devices[i].power = false;
    devices[i].speed = 0;
  }
}

void device_control_set(uint8_t id, bool power, uint8_t speed) {
  if (id >= 4) return;

  devices[id].power = power;

  if (devices[id].type == DEVICE_FAN) {
    devices[id].speed = power ? speed : 0;
  }
}

DeviceState device_control_get(uint8_t id) {
  if (id >= 4) {
    DeviceState dummy = { DEVICE_LIGHT, false, 0 };
    return dummy;
  }
  return devices[id];
}

#include <Arduino.h>
#include <WiFi.h>
#include <WebServer.h>
#include "webserver.h"
#include "device_control.h"

static WebServer server(80);

static void handle_status() {
  String json = "{ \"devices\":[";
  for (int i = 0; i < 4; i++) {
    DeviceState st = device_control_get(i);
    json += "{";
    json += "\"id\":" + String(i) + ",";
    json += "\"power\":" + String(st.power ? "true" : "false") + ",";
    json += "\"speed\":" + String(st.speed);
    json += "}";
    if (i < 3) json += ",";
  }
  json += "] }";

  server.send(200, "application/json", json);
}

static void handle_control() {
  if (!server.hasArg("id") || !server.hasArg("power")) {
    server.send(400, "text/plain", "Missing params");
    return;
  }

  int id = server.arg("id").toInt();
  bool p = server.arg("power") == "1";
  int speed = server.hasArg("speed") ? server.arg("speed").toInt() : 0;

  device_control_set(id, p, speed);
  server.send(200, "application/json", "{\"ok\":true}");
}

void webserver_init() {
  server.on("/status", HTTP_GET, handle_status);
  server.on("/control", HTTP_GET, handle_control);
  server.begin();
}

void webserver_loop() {
  server.handleClient();
}

#include <Arduino.h>
#include <WiFi.h>
#include <WebServer.h>
#include <DNSServer.h>
#include <Preferences.h>
#include <LittleFS.h>
#include "wifi_manager.h"


static WebServer server(80);
static DNSServer dnsServer;
static Preferences prefs;

const char* AP_SSID = "SmartSetup";
const byte DNS_PORT = 53;
const char* PREF_NAMESPACE = "smarthome";
const char* PREF_SSID = "ssid";
const char* PREF_PWD  = "pwd";

static bool runningAP = false;

// helper: determine common mime types
static String getContentType(const String& path) {
  if (path.endsWith(".htm") || path.endsWith(".html")) return "text/html";
  if (path.endsWith(".css")) return "text/css";
  if (path.endsWith(".js")) return "application/javascript";
  if (path.endsWith(".png")) return "image/png";
  if (path.endsWith(".jpg")) return "image/jpeg";
  if (path.endsWith(".ico")) return "image/x-icon";
  return "text/plain";
}

// serve a file from LittleFS
static bool streamFileIfExists(const String& path) {
  String p = path;
  if (p == "/") p = "/index.html";
  if (!LittleFS.exists(p)) return false;
  File f = LittleFS.open(p, "r");
  if (!f) return false;
  server.streamFile(f, getContentType(p));
  f.close();
  return true;
}

// HTTP handlers
static void handleRoot() {
  if (!streamFileIfExists("/index.html")) {
    server.send(500, "text/plain", "index.html not found on device filesystem");
  }
}

static void handleFile() {
  String path = server.uri();
  if (!streamFileIfExists(path)) {
    server.send(404, "text/plain", "Not found");
  }
}

static void handleScan() {
  int n = WiFi.scanNetworks();
  String json = "[";
  for (int i = 0; i < n; i++) {
    json += "{";
    json += "\"ssid\":\"" + String(WiFi.SSID(i)) + "\",";
    json += "\"rssi\":" + String(WiFi.RSSI(i)) + ",";
    json += "\"open\":" + String(WiFi.encryptionType(i) == WIFI_AUTH_OPEN ? "true" : "false");
    json += "}";
    if (i < n - 1) json += ",";
  }
  json += "]";
  server.send(200, "application/json", json);
}

static void handleSave() {
  if (server.hasArg("ssid") && server.hasArg("pwd")) {
    String ssid = server.arg("ssid");
    String pwd  = server.arg("pwd");
    prefs.begin(PREF_NAMESPACE, false);
    prefs.putString(PREF_SSID, ssid);
    prefs.putString(PREF_PWD, pwd);
    prefs.end();
    server.send(200, "text/html", "<html><body>Saved. Rebooting...</body></html>");
    delay(800);
    ESP.restart();
  } else {
    server.send(400, "text/plain", "Missing args");
  }
}

void startAP() {
  WiFi.mode(WIFI_AP);
  WiFi.softAP(AP_SSID);
  IPAddress apIP = WiFi.softAPIP(); // usually 192.168.4.1
  dnsServer.start(DNS_PORT, "*", apIP);

  // register endpoints
  server.on("/", HTTP_GET, handleRoot);
  server.on("/scan", HTTP_GET, handleScan);
  server.on("/save", HTTP_POST, handleSave);

  // try to serve static files (css/js, images) from LittleFS
  server.onNotFound(handleFile);

  server.begin();
  runningAP = true;
}

void stopAP() {
  server.stop();
  dnsServer.stop();
  WiFi.softAPdisconnect(true);
  runningAP = false;
}

void trySTAConnect(const String& ssid, const String& pwd) {
  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid.c_str(), pwd.c_str());
  unsigned long start = millis();
  const unsigned long timeout = 15000; // 15s
  while (millis() - start < timeout) {
    if (WiFi.status() == WL_CONNECTED) break;
    delay(200);
  }
}

void wifi_manager_init() {
  // mount LittleFS (will host index.html and assets)
  if (!LittleFS.begin()) {
    Serial.println("LittleFS mount failed");
  } else {
    Serial.println("LittleFS mounted");
  }

  prefs.begin(PREF_NAMESPACE, true); // read-only
  String saved_ssid = prefs.getString(PREF_SSID, "");
  String saved_pwd  = prefs.getString(PREF_PWD, "");
  prefs.end();

  if (saved_ssid.length() > 0) {
    trySTAConnect(saved_ssid, saved_pwd);
    if (WiFi.status() == WL_CONNECTED) {
      Serial.print("Connected on STA mode: ");
      Serial.println(WiFi.localIP());
      return;
    }
  }
  // Fallback to AP mode / captive portal
  startAP();
}

void wifi_manager_loop() {
  if (runningAP) {
    dnsServer.processNextRequest();
    server.handleClient();
  } else {
    // Monitor STA connection; restart AP if lost
    if (WiFi.status() != WL_CONNECTED) {
      startAP();
    }
  }
}

void wifi_manager_clear_credentials() {
  prefs.begin(PREF_NAMESPACE, false);
  prefs.remove(PREF_SSID);
  prefs.remove(PREF_PWD);
  prefs.end();
}
